1.2校验function

        function addValid(){
            window.validater.add('age', 'numeric', {
                oi: true,
                min: 1,
                notim: '年龄必须为整数',
                minm: '年龄必须大于0'
            });
            window.validater.add('age', 'required', {
                m:'年龄不能为空'
            });
        }
        function submitForm(){
            //验证表单
            window.validater.validAllElement();
        }
